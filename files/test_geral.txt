  /*
  * Metodo que multiplica x e y
  */
  public int multiplica(int x, int y){
  	return x * y;
  }
  
  public int qtLoopsWhile(int i){
  	int contador = 0;
  	while(true){
  		System.out.print("LOOP WHILE: " + contador);
  		contador ++;
  		if(contador == i){ 
  			break;
  		}
  	}
  	return contador;
  }
  
  public int qtLoopsFor(int i){
  	for(int j = 0; j < i; j++){
  		System.out.print("LOOP FOR: " + j);
  	}
  	return i;
  }

          	break;

        	System.out.printf("%.2f - %.2f = %.2f", numero1, numero2, numero1 - numero2);

        case '*':
       		System.out.printf("%.2f * %.2f = %.2f", numero1, numero2, numero1 * numero2);

            System.out.printf("Você digitou uma operação inválida.");        
    }
  }
